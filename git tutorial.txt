cara buat branch
git clone [alamat ssh git yg mau diclone]
ls
cd [nama file]
git checkout -b [nama branch yg mau dibuat]  : buat branch baru
git add  .
git commit -m "first commit"
git push -u origin [nama branch]

cara pindah branch
git checkout [nama branch yg ingin dituju]

cara lihat daftar branch
git checkout


cara ngepull atau mau ambil kodingan dari branch
git branch
git pull origin [nama branch yang ingin dipull]


cara cancel pull yg sudah di merging
git merge --abort

cara restart kodingan git setelah di pull (biar balik lagi)
git restart --hard
